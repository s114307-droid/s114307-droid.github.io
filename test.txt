<script>
    const cat = document.getElementById("cat");
    let x = 200;
    let y = 200;
    const speed = 10;

    document.addEventListener("keydown", (e) => {
        switch (e.key) {
            case "ArrowUp":
                y -= speed;
                break;
            case "ArrowDown":
                y += speed;
                break;
            case "ArrowLeft":
                x -= speed;
                cat.style.transform = "scaleX(-1)"; // 左右翻轉
                break;
            case "ArrowRight":
                x += speed;
                cat.style.transform = "scaleX(1)";
                break;
        }
        cat.style.left = x + "px";
        cat.style.top = y + "px";
    });
</script>
